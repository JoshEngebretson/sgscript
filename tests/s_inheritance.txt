var printerIface = 
{
	print = function( text ){ print( text ); },
};
var postfixPrinterIface = 
{
	print = function( this, text ){ print( text, this.postfix ); },
	postfix = '[postfix]',
};
var questionPrinter = class( { postfix = '?' }, postfixPrinterIface );
questionPrinter.print( "wat" );
