print( "--== ARRAY ==--\n" );
arr = [ "one", "two", "three" ];
foreach( x : arr )
	print( x, " = ", arr[x], "\n" );
foreach( x : arr )
{
	if( x == 1 ) continue;
	print( x );
	if( x >= 1 ) break;
}
print( "\n" );

print( "--== DICT ==--\n" );
map = { "1" = "one", "2" = "two", "3" = "three" };
map[ "1.5" ] = "one and a half";
foreach( x : map )
	print( x, " = ", map[x], "\n" );
