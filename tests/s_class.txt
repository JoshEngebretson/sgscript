var myInfoIface = 
{
	greetMe = function( this ){ print( "Hello, " $ this._name $ "!\n" ); },
	setMyName = function( this, name ){ this._name = name; },
};
myInfo = class( {}, myInfoIface );
print( "Created a ", myInfo, " object with this interface: ", myInfo._super, "\n" );
myInfo.setMyName( "SGScript" );
myInfo.greetMe();
