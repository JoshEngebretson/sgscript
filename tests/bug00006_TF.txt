global ERRORS, ERRLINE;// basic leveling errorERRORS = "";eval( 'amazing  // 1amazing;' );    // 2testEqual( ERRLINE $ ',' $ ERRORS, "2,[E:Missing operators or separators.]" );// extended leveling error 1ERRORS = "";eval( '(  // 1function  // 2(         // 3)         // 4{         // 5}         // 6onenter   // 7)         // 8;' );     // 9testEqual( ERRLINE $ ',' $ ERRORS, "7,[E:Missing operators or separators.]" );// extended leveling error 2ERRORS = "";eval( 'x  // 1=         // 2function  // 3(         // 4)         // 5{         // 6}         // 7onenter   // 8;' );     // 9testEqual( ERRLINE $ ',' $ ERRORS, "8,[E:Missing operators or separators.]" );
